# 两两交换链表中的节点

## JS题解代码

```
var swapPairs = function(head) {
    let dummy = new ListNode();
    dummy.next = head;
    let curr = dummy;
    while(curr.next !== null && curr.next.next !== null){
        let n1 = curr.next;
        let n2 = curr.next.next;
        curr.next = n2;
        n1.next = n2.next;
        n2.next = n1;
        curr = n1;
    }
    return dummy.next;
};
```
